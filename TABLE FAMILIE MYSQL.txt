CREATE TABLE `presentmanager`.`familie` (
  

`idFamilie` INT UNSIGNED NOT NULL AUTO_INCREMENT COMMENT '',
  
`Vorname` VARCHAR(45) NULL COMMENT '',
 
 `Nachname` VARCHAR(45) NULL COMMENT '',
 
 `Geburtsdatum` DATE NULL DEFAULT NULL COMMENT '',
 
 `Age` TINYINT(3) NULL COMMENT '',
 
 `Hobbys` VARCHAR(45) NULL DEFAULT 'default' COMMENT '',
 
 `Image` BLOB NULL COMMENT '',
 
 PRIMARY KEY (`idFamilie`)  COMMENT '')

ENGINE=INNODB;
SELECT Vorname,Geburtsdatum, CURDATE(),
 TIMESTAMPDIFF(YEAR,Geburtsdatum,CURDATE()) AS'Age in year'
FROM presentmanager.familie ORDER BY idFamilie;
